package com.example.kingdom_king;

import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.widget.AppCompatButton;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class PagB_Login_Activity extends BaseVoiceActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_pag2_login);

        // Ajuste de márgenes para el diseño EdgeToEdge
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        // Botón de inicio de sesión principal (ID: btnLogin)
        AppCompatButton btnLogin = findViewById(R.id.btnLogin);
        btnLogin.setOnClickListener(v -> irAHome());

        // Botones de inicio de sesión con Apple y Google
        AppCompatButton btnApple = findViewById(R.id.button_loggin_apple);
        AppCompatButton btnGoogle = findViewById(R.id.button_loggin_google);

        btnApple.setOnClickListener(v -> {
            Intent intent = new Intent(Intent.ACTION_VIEW, Uri.parse("https://appleid.apple.com/"));
            startActivity(intent);
        });

        btnGoogle.setOnClickListener(v -> {
            Intent intent = new Intent(Intent.ACTION_VIEW, Uri.parse("https://accounts.google.com/"));
            startActivity(intent);
        });

        // IMPORTANTE: Iniciamos la escucha del micrófono al entrar en la pantalla
        comprobarPermisoYEmpezar();
    }

    /**
     * Navega a la pantalla principal (PagC) y cierra la actual
     */
    private void irAHome() {
        Intent intent = new Intent(this, PagC_Home_Activity.class);
        startActivity(intent);
        finish();
    }

    /**
     * Se activa cuando BaseVoiceActivity reconoce un comando de voz
     */
    @Override
    protected void onVoiceCommand(String comando) {
        // Normalizamos el comando para detectar variaciones con o sin tilde
        String comandoNormalizado = comando.toLowerCase().trim();

        if (comandoNormalizado.contains("iniciar sesión") || comandoNormalizado.contains("iniciar sesion")) {
            hablar("Iniciando sesión, bienvenido Jarl");
            irAHome();
        }
    }
}






*********************EL ANDROID MANIFEST******************************


He limpiado tu AndroidManifest.xml para organizar el flujo de la aplicación correctamente y evitar que se creen múltiples iconos en el menú del móvil.
Cambios realizados:
1.
Permisos de Audio: He añadido los permisos de RECORD_AUDIO e INTERNET necesarios para que el Speech To Text funcione.
2.
Un solo Launcher: He dejado únicamente la PagA_Splash_Screen_Activity con el filtro de LAUNCHER. Esto asegura que la aplicación siempre empiece por el Splash Screen y no cree 10 iconos diferentes.
3.
Exportación de Actividades: He puesto android:exported="false" en todas las demás pantallas para mejorar la seguridad, ya que solo el Splash Screen debe ser accesible desde fuera.
4.
MainActivity Comentada: He comentado la declaración de MainActivity tal como solicitaste.
